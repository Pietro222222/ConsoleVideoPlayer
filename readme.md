# how it works

video.mp4 <br> 
   | 
<br>
   | -> video.gif -> play gif through viu
<br>
   |
<br>   
| -> video.mp3 \
<br>                   \-> play sound

cache: $HOME/.console_player/cache/{video-hash}/
                                    -> video.mp3
                                    -> video.gif
# How to use

./consoleplayer video.mp4
